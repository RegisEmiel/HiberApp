use hiberapp;

CREATE TABLE products (id integer AUTO_INCREMENT PRIMARY KEY, title varchar(255), price int);

CREATE TABLE shopcarts (id integer AUTO_INCREMENT PRIMARY KEY, cart_number varchar(255));

CREATE TABLE users (id integer AUTO_INCREMENT PRIMARY KEY, shopcart_id integer, nik_name varchar(255), email varchar(255), foreign key (shopcart_id) references shopcarts (id));

CREATE TABLE shopcart_details (shopcart_id integer, product_id integer, foreign key (shopcart_id) references shopcarts (id), foreign key (product_id) references products (id));
